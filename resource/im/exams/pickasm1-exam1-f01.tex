\begin{minipage}{3in}
\begin{ccode}
\begin{alltt}
int fun1(int a, int b)
\verb:{:
    if (a < b)
        return a;
    else
        return b;
\verb:}:
\end{alltt}
\begin{alltt}
int fun2(int a, int b)
\verb:{:
    if (b < a)
        return b;
    else
        return a;
\verb:}:
\end{alltt}
\begin{alltt}
int fun3(int a, int b)
\verb:{:
    unsigned ua = (unsigned) a;
    if (ua < b)
        return b;
    else
        return ua;
\verb:}:
\end{alltt}
\end{ccode}
\end{minipage}
\begin{minipage}{3in}
\begin{scode}
\begin{alltt}
  pushl %ebp{\em\scriptsize }
  movl %esp,%ebp{\em\scriptsize }

  movl 12(%ebp),%eax{\em\scriptsize }
  cmpl %eax,%edx{\em\scriptsize }
  jge .L9{\em\scriptsize }
  movl %edx,%eax{\em\scriptsize }
.L9:{\em\scriptsize }
  movl %ebp,%esp{\em\scriptsize }
  popl %ebp{\em\scriptsize }
  ret{\em\scriptsize }
\end{alltt}
\end{scode}
\end{minipage}
